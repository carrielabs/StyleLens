    function renderInteractiveChart() {

      const svg = document.getElementById('interactive-svg-canvas');
      
      if (!container || !svg) return;

      const width = container.clientWidth;
      const height = container.clientHeight;

      // 1. 设置 SVG viewBox 为 1:1 绝对物理尺寸，完全消除缩放变异
      svg.setAttribute('viewBox', `0 0 ${width} ${height}`);

      // 留出图表的四周边界线 padding
      const padding = { top: 15, right: 10, bottom: 24, left: 10 };
      const plotWidth = width - padding.left - padding.right;
      const plotHeight = height - padding.top - padding.bottom;

      // 坐标映射工具
      const getX = (idx) => padding.left + (idx / (chartData.length - 1)) * plotWidth;
      const getYViews = (val) => padding.top + plotHeight - (val / 55000) * plotHeight;
      const getYRates = (val) => padding.top + plotHeight - (val / 3.0) * plotHeight;

      // 计算每个点的物理坐标位置
      const points = chartData.map((d, idx) => ({
        x: getX(idx),
        yViews: getYViews(d.views),
        yRate: getYRates(d.rate),
        ...d
      }));

      // 2. 动态拼装曝光折线与渐变区域填充的 Path Data
      let pathD = '';
      let fillD = '';
      let ratePathD = '';

      points.forEach((pt, idx) => {
        if (idx === 0) {
          pathD = `M ${pt.x} ${pt.yViews}`;
          fillD = `M ${pt.x} ${padding.top + plotHeight} L ${pt.x} ${pt.yViews}`;
          ratePathD = `M ${pt.x} ${pt.yRate}`;
        } else {
          pathD += ` L ${pt.x} ${pt.yViews}`;
          fillD += ` L ${pt.x} ${pt.yViews}`;
          ratePathD += ` L ${pt.x} ${pt.yRate}`;
        }
        if (idx === points.length - 1) {
          fillD += ` L ${pt.x} ${padding.top + plotHeight} Z`;
        }
      });

      // 应用 Path 路径数据
      document.getElementById('svg-fill-area').setAttribute('d', fillD);
      document.getElementById('svg-views-path').setAttribute('d', pathD);
      document.getElementById('svg-rates-path').setAttribute('d', ratePathD);

      // 3. 绘制节点圆点：通过 SVG 矢量保证渲染，防止任何横向畸变
      const dotsGroup = document.getElementById('svg-dots-group');
      dotsGroup.innerHTML = ''; // 清空重新构建

      points.forEach((pt, idx) => {
        // 创建组合容器
        const dotG = document.createElementNS('http://www.w3.org/2000/svg', 'g');
        dotG.setAttribute('class', 'interactive-dot-group cursor-pointer');
        dotG.setAttribute('data-index', idx);

        // A. 底层透明触控热区（扩大鼠标交互面积，防抖动）
        const touchCircle = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
        touchCircle.setAttribute('cx', pt.x);
        touchCircle.setAttribute('cy', pt.yViews);
        touchCircle.setAttribute('r', 16);
        touchCircle.setAttribute('fill', 'transparent');
        touchCircle.setAttribute('pointer-events', 'all');

        // B. 节点真实等比同心圆：外描边（黑）+ 填充芯（荧光绿）+ 绝对等比正圆半径
        const coreCircle = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
        coreCircle.setAttribute('id', `chart-dot-${idx}`);
        coreCircle.setAttribute('cx', pt.x);
        coreCircle.setAttribute('cy', pt.yViews);
        coreCircle.setAttribute('class', 'interactive-dot');
        
        if (pt.isPeak) {
          // 峰值高亮：外环荧光、内嵌雪白，自带高斯光晕 filter
          coreCircle.setAttribute('r', 5);
          coreCircle.setAttribute('fill', '#ffffff');
          coreCircle.setAttribute('stroke', '#7CFF44');
          coreCircle.setAttribute('stroke-width', '2.5');
          coreCircle.setAttribute('filter', 'url(#peakGlowFilter)');
        } else {
          // 普通节点
          coreCircle.setAttribute('r', 3.5);
          coreCircle.setAttribute('fill', '#070A08');
          coreCircle.setAttribute('stroke', '#7CFF44');
          coreCircle.setAttribute('stroke-width', '1.5');
        }

        dotG.appendChild(touchCircle);
        dotG.appendChild(coreCircle);
        dotsGroup.appendChild(dotG);
      });

      // 4. 高级磁吸交互：对整个 SVG 容器区域进行光标距离演算
      const guideLine = document.getElementById('svg-guide-line');
      const tooltip = document.getElementById('chart-tooltip');

      // 核心交互：追踪鼠标相对于图表的 X 物理像素位置，自动匹配最近节点 (磁吸对齐)
      function handleChartHover(clientX) {
        const rect = container.getBoundingClientRect();
        const relativeX = clientX - rect.left;

        if (relativeX < padding.left - 10 || relativeX > padding.left + plotWidth + 10) {
          resetHoverState();
          return;
        }

        // 计算当前位置落在哪个数据节点区间
        const calculatedIndex = Math.round(((relativeX - padding.left) / plotWidth) * (chartData.length - 1));
        const activeIdx = Math.max(0, Math.min(chartData.length - 1, calculatedIndex));
        const activePt = points[activeIdx];

        // A. 悬浮垂直指示线对齐
        guideLine.setAttribute('x1', activePt.x);
        guideLine.setAttribute('x2', activePt.x);
        guideLine.setAttribute('y1', padding.top);
        guideLine.setAttribute('y2', padding.top + plotHeight);
        guideLine.classList.remove('opacity-0');

        // B. 圆点缩放激活状态变换
        points.forEach((pt, idx) => {
          const dot = document.getElementById(`chart-dot-${idx}`);
          if (!dot) return;
          if (idx === activeIdx) {
            // 被 Hover 选中的点平滑放大，加深边缘
            dot.setAttribute('r', pt.isPeak ? 7 : 5.5);
            dot.setAttribute('stroke-width', pt.isPeak ? '3.5' : '3');
            dot.setAttribute('fill', pt.isPeak ? '#ffffff' : '#7CFF44');
          } else {
            // 恢复普通点
            if (pt.isPeak) {
              dot.setAttribute('r', 5);
              dot.setAttribute('stroke-width', '2.5');
              dot.setAttribute('fill', '#ffffff');
            } else {
              dot.setAttribute('r', 3.5);
              dot.setAttribute('stroke-width', '1.5');
              dot.setAttribute('fill', '#070A08');
            }
          }
        });

        // C. 信息高密度 Tooltip 渲染
        tooltip.innerHTML = `
          <div class="text-[10px] font-mono text-cyber-muted mb-1">${activePt.date} 数据快照</div>
          <div class="text-xs font-bold text-white mb-2 truncate">${activePt.desc}</div>
          <div class="flex justify-between items-center text-[11px] mb-1">
            <span class="text-cyber-muted">曝光流量 (PV)</span>
            <span class="font-mono text-cyber-volt font-bold">${activePt.views.toLocaleString()}</span>
          </div>
          <div class="flex justify-between items-center text-[11px]">
            <span class="text-cyber-muted">平均转化效率</span>
            <span class="font-mono text-white font-semibold">${activePt.rate}%</span>
          </div>
        `;

        // 计算 Tooltip 智能浮动偏移（防止超出右边界）
        let tooltipLeft = activePt.x - 88; // 宽度的一半
        if (tooltipLeft < 5) tooltipLeft = 5;
        if (tooltipLeft + 180 > width) tooltipLeft = width - 180;

        tooltip.style.left = `${tooltipLeft}px`;
        tooltip.style.top = `${activePt.yViews - 82}px`;
        tooltip.style.opacity = '1';
      }

      function resetHoverState() {
        guideLine.classList.add('opacity-0');
        tooltip.style.opacity = '0';
        // 还原所有点
        points.forEach((pt, idx) => {
          const dot = document.getElementById(`chart-dot-${idx}`);
          if (!dot) return;
          if (pt.isPeak) {
            dot.setAttribute('r', 5);
            dot.setAttribute('stroke-width', '2.5');
            dot.setAttribute('fill', '#ffffff');
          } else {
            dot.setAttribute('r', 3.5);
            dot.setAttribute('stroke-width', '1.5');
            dot.setAttribute('fill', '#070A08');
          }
        });
      }

      // 挂载交互触发器
      container.addEventListener('mousemove', (e) => handleChartHover(e.clientX));
      container.addEventListener('mouseleave', resetHoverState);
      container.addEventListener('touchmove', (e) => {
        if (e.touches && e.touches[0]) {
          handleChartHover(e.touches[0].clientX);
        }
      });
      container.addEventListener('touchend', resetHoverState);
    }

    // ================= STATIC TEMPLATE INITIALIZATION =================
    document.addEventListener("DOMContentLoaded", () => {
      // 1. 初始化富交互时序折线图
      renderInteractiveChart();
      window.addEventListener('resize', renderInteractiveChart);

      // 2. 自动为所有 data-editable="true" 的元素增加 html5 contenteditable 属性，使其可以直接在浏览器编辑
      const editables = document.querySelectorAll('[data-editable="true"]');
      editables.forEach(elem => {
        elem.setAttribute("contenteditable", "true");
        elem.setAttribute("spellcheck", "false");
        
        // 聚焦时选定全部文本
        elem.addEventListener("focus", (e) => {
          setTimeout(() => {
            const range = document.createRange();
            range.selectNodeContents(elem);
            const sel = window.getSelection();
            sel.removeAllRanges();
            sel.addRange(range);
          }, 50);
        });
      });
    });
