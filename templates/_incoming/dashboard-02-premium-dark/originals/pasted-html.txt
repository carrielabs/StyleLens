    // Custom In-place text editing controller
    function setupInPlaceEditing() {
      document.querySelectorAll('[data-editable="true"]').forEach(elem => {
        elem.addEventListener('dblclick', function() {
          // Verify current operational state
          if (!document.body.classList.contains('edit-mode')) {
            // Auto switch to edit mode to lower barrier
            toggleEditMode(true);
          }

          const originalText = this.innerText;
          const input = document.createElement('textarea');
          input.value = originalText;
          input.className = "bg-zinc-900 border border-brand text-zinc-100 p-1.5 rounded w-full text-xs font-medium focus:outline-none focus:ring-1 focus:ring-brand";
          
          // Replicate sizing to minimize layout shifts
          input.style.height = `${this.offsetHeight + 10}px`;

          const saveChanges = () => {
            const val = input.value.trim();
            if (val) {
              this.innerText = val;
            } else {
              this.innerText = originalText;
            }
            input.remove();
            this.style.display = '';
            showToast("单元数据已修改成功");
          };

          input.addEventListener('blur', saveChanges);
          input.addEventListener('keydown', (e) => {
            if (e.key === 'Enter' && !e.shiftKey) {
              e.preventDefault();
              saveChanges();
            }
          });

          this.style.display = 'none';
          this.parentNode.insertBefore(input, this);
          input.focus();
        });
      });
    }

    function initWorldMap() {
      const canvas = document.getElementById('worldMapCanvas');
      if (!canvas) return;
      const ctx = canvas.getContext('2d');

      // Sync sizing to bounding container
      const resizeMap = () => {
        canvas.width = canvas.parentElement.clientWidth;
        canvas.height = canvas.parentElement.clientHeight;
        drawMapContent();
      };

      // Custom pseudo world-map coordinate generator to represent Spotify presence clusters
      const clusters = [
        // LatAm (Heavy Green cluster)
        { cx: 0.32, cy: 0.72, density: 42, label: "巴西 (Brazil)", size: 30, power: 1.0 },
        { cx: 0.28, cy: 0.62, density: 15, label: "哥伦比亚", size: 18, power: 0.6 },
        // North America

        { cx: 0.25, cy: 0.25, density: 12, label: "加拿大", size: 20, power: 0.5 },
        // Europe

        { cx: 0.53, cy: 0.33, density: 25, label: "东欧", size: 22, power: 0.7 },
        // Asia / Oceania
        { cx: 0.78, cy: 0.42, density: 20, label: "东南亚区", size: 24, power: 0.6 },
        { cx: 0.82, cy: 0.75, density: 18, label: "澳大利亚", size: 22, power: 0.75 }
      ];

      function drawMapContent() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        // 1. Draw a clean, sci-fi analytical coordinate grid lines
        ctx.strokeStyle = 'rgba(63, 63, 70, 0.15)';
        ctx.lineWidth = 1;
        const gridGap = 40;
        for (let x = 0; x < canvas.width; x += gridGap) {
          ctx.beginPath();
          ctx.moveTo(x, 0);
          ctx.lineTo(x, canvas.height);
          ctx.stroke();
        }
        for (let y = 0; y < canvas.height; y += gridGap) {
          ctx.beginPath();
          ctx.moveTo(0, y);
          ctx.lineTo(canvas.width, y);
          ctx.stroke();
        }

        // 2. Draw high-tech analytical particles representing countries
        clusters.forEach(cluster => {
          const mapX = cluster.cx * canvas.width;
          const mapY = cluster.cy * canvas.height;

          // Ambient halo glow beneath high-value clusters
          const glowGrad = ctx.createRadialGradient(mapX, mapY, 2, mapX, mapY, cluster.size * 1.8);
          glowGrad.addColorStop(0, currentAccentColor.replace(')', ', 0.25)').replace('rgb', 'rgba'));
          glowGrad.addColorStop(1, 'rgba(0, 0, 0, 0)');
          ctx.fillStyle = glowGrad;
          ctx.beginPath();
          ctx.arc(mapX, mapY, cluster.size * 1.8, 0, Math.PI * 2);
          ctx.fill();

          // Dotted grid cluster structure representing coordinates
          for (let i = 0; i < cluster.density; i++) {
            const angle = Math.random() * Math.PI * 2;
            // Standard deviation distribution for density visual
            const r = (Math.pow(Math.random(), 1.5)) * cluster.size;
            const ptX = mapX + Math.cos(angle) * r;
            const ptY = mapY + Math.sin(angle) * r;

            // Varying opacity to look complex and dynamic
            ctx.fillStyle = `rgba(${hexToRgb(currentAccentColor)}, ${0.1 + Math.random() * cluster.power})`;
            ctx.beginPath();
            ctx.arc(ptX, ptY, 1.5 + Math.random() * 1.5, 0, Math.PI * 2);
            ctx.fill();
          }

          // Render coordinate labels
          ctx.font = 'bold 8px monospace';
          ctx.fillStyle = 'rgba(161, 161, 170, 0.6)';
          ctx.fillText(`LOC-${cluster.label.substring(0,2).toUpperCase()}`, mapX - 15, mapY - cluster.size - 4);
        });
      }

      window.addEventListener('resize', resizeMap);
      resizeMap();
    }

    // Helper: Hex translation to raw RGB values
    function hexToRgb(hex) {
      const shorthandRegex = /^#?([a-f\d])([a-f\d])([a-f\d])$/i;
      const fullHex = hex.replace(shorthandRegex, (m, r, g, b) => r + r + g + g + b + b);
      const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(fullHex);
      return result ? `${parseInt(result[1], 16)}, ${parseInt(result[2], 16)}, ${parseInt(result[3], 16)}` : '29, 185, 84';
    }
